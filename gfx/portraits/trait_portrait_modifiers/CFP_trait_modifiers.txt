cfp_eyeglasses = {
	cfp_eyeglasses = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = special_headgear_eyeglasses
				template = eyeglasses
				range = { 0 1 } # For the randomness to work correctly
			}
		}
		traits = {
			whole_of_body
			administrator
		}
		trigger = {
			trigger_if = {
				limit = { exists = root }
				age >= 40
				is_alive = yes
				current_date >= 1286.1.1
				NOT = { has_game_rule = never_cfp_trait_accessories }
				OR = {
					AND = {
						learning >= 12
						portrait_has_trait_trigger = { TRAIT = whole_of_body }
					}
					AND = {
						stewardship >= 12
						portrait_has_trait_trigger = { TRAIT = administrator }
					}
				}
				NOR = {
					portrait_has_trait_trigger = { TRAIT = disfigured }
					portrait_has_trait_trigger = { TRAIT = blind }
					portrait_indian_clothing_trigger = yes
					portrait_african_clothing_trigger = yes
					portrait_steppe_clothing_trigger = yes
					cfp_portrait_norse_clothing_trigger = yes
					cfp_portrait_turkic_clothing_trigger = yes
				}
			}
		}
	}
}
