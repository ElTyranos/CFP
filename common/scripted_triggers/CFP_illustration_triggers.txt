has_graphical_norman_culture_group_trigger = {
	OR = {
		has_culture = culture:english
		has_culture = culture:norman
		AND = {
			culture = {	has_cultural_pillar = heritage_frankish }
			capital_county.kingdom = title:k_england
			NOR = {
				has_culture = culture:french
				has_culture = culture:frankish
				has_culture = culture:occitan
			}
		}
	}
}

has_graphical_goidelic_culture_group_trigger = {
	OR = {
		has_culture = culture:irish
		has_culture = culture:gaelic
		culture = {	has_cultural_pillar = heritage_goidelic }
	}
}

has_graphical_andalusian_culture_group_trigger = {
	OR = {
		has_culture = culture:andalusian
		AND = {
			faith.religion = religion:islam_religion
			capital_county.kingdom = title:k_andalusia
		}
		AND = {
			culture = { has_cultural_pillar = heritage_iberian }
			capital_county.kingdom = title:k_andalusia
		}
	}
}
