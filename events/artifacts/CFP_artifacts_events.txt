# A Burmese traveler might forget his Kris Dagger if you invite him
cfp_artifact.0001 = {
	type = character_event
	title = cfp_artifact.0001.t
	desc = cfp_artifact.0001.desc
	theme = court
	left_portrait = {
		character = root
		animation = idle
	}
	right_portrait = {
		character = scope:traveler
		animation = beg
	}
	
	trigger = {
		NOR = {
			has_character_flag = had_event_cfp_artifact_0001
			has_culture = culture:burmese
			has_culture = culture:mon
			location = { geographical_region = world_europe_north }
			location = { geographical_region = world_africa }
			has_game_rule = never_cfp_artifacts # Disable if rule is OFF
		}
	}
	
	immediate = {
		create_character = {
			gender = male
			age = { 35 50 }
			learning = { 8 12 }
			intrigue = { 6 8 }
			trait = education_intrigue_3
			faith = faith:theravada
			culture = culture:burmese
			location = root.capital_province
			save_scope_as = traveler
			dynasty = none
		}
		add_character_flag = {
			flag = had_event_cfp_artifact_0001
			days = 7300
		}
	}
	
	## Go away!
	option = {
		name = cfp_artifact.0001.a
		debug_log = "CFP_Debug: traveler was denied entry in cfp_artifact.0001"
		
		stress_impact = {
			compassionate = medium_stress_impact_gain
			generous = medium_stress_impact_gain
			gregarious = minor_stress_impact_gain
		}
		
		hidden_effect = {
			scope:traveler = {
				death = { death_reason = death_vanished	}
			}
		}
		
		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_compassion = -1
				ai_honor = -1
			}
			modifier = {
				faith = {
				faith_hostility_level = {
					target = scope:traveler.faith
					value = faith_astray_level
				}
			}
			add = 50
			}
			modifier = {
				faith = {
				faith_hostility_level = {
					target = scope:traveler.faith
					value = faith_hostile_level
				}
			}
			add = 100
			}
			modifier = {
				faith = {
				faith_hostility_level = {
					target = scope:traveler.faith
					value = faith_evil_level
				}
			}
			add = 200
			}
		}
		
	}
	
	## Jail
	option = {
		name = cfp_artifact.0001.b
		debug_log = "CFP_Debug: traveler was imprisoned in cfp_artifact.0001"
		
		add_dread = minor_dread_gain
		
		stress_impact = {
			sadistic = major_stress_impact_loss
			wrathful = minor_stress_impact_loss
			lunatic_1 = minor_stress_impact_loss
			lunatic_genetic = minor_stress_impact_loss
			paranoid = medium_stress_impact_loss
		}
		
		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_compassion = -1
				ai_zeal = 1
			}
			# modifier = {
				# OR = {
					# has_trait = zealous
				# }
				# add = 100
			# }
			modifier = {
				OR = {
					has_trait = sadistic
					has_trait = wrathful
					has_trait = callous
				}
				add = 200
			}
		}

		imprison_character_effect = {
			TARGET = scope:traveler
			IMPRISONER = root
		}
	}
	
	## Sleep in the stables
	option = {
		name = cfp_artifact.0001.c
		
		remove_short_term_gold = tiny_gold_value
		
		stress_impact = {
			paranoid = medium_stress_impact_gain
			callous = minor_stress_impact_gain
			sadistic = medium_stress_impact_gain
			zealous = major_stress_impact_gain
			greedy = minor_stress_impact_gain
		}
				
		ai_chance = {
			base = 50

			ai_value_modifier = {
				ai_compassion = 1
				ai_honor = 1
				ai_zeal = -2
				ai_greed = -1
				min = -45
			}
			modifier = {
				OR = {
					has_trait = greedy
				}
				add = -100
			}
			modifier = {
				faith = {
				faith_hostility_level = {
					target = scope:traveler.faith
					value = faith_hostile_level
				}
			}
			add = -50
			}
			modifier = {
				faith = {
				faith_hostility_level = {
					target = scope:traveler.faith
					value = faith_evil_level
				}
			}
			add = -100
			}
		}
		
		hidden_effect = {
			random_list = {
				60 = {
					debug_log = "CFP_Debug: traveler forgot his dagger in the stables cfp_artifact.0001"
					if = { 
						limit = { NOT = { any_character_artifact = { has_variable = cfp_kris_dagger_var } } } 
						trigger_event = {
							id = cfp_artifact.0002
							days = 1
						}
					}
					else = {
						debug_log = "CFP_Debug: traveler slept in the stables in cfp_artifact.0001"
						trigger_event = {
							id = cfp_artifact.0003
							days = 1
						}
					}
				}
				40 = {
					debug_log = "CFP_Debug: traveler slept in the stables in cfp_artifact.0001"
					trigger_event = {
						id = cfp_artifact.0003
						days = 1
					}
				}
			}
		}
	}
}

# A local sculptor asks you to kickstart him so he can prove his talent
cfp_artifact.0005 = {
	type = character_event
	title = cfp_artifact.0005.t
	desc = cfp_artifact.0005.desc
	theme = court
	left_portrait = {
		character = root
		animation = idle
	}
	right_portrait = {
		character = scope:sculptor
		animation = beg
	}
	
	trigger = {
		faith = { religion_tag = christianity_religion }
		NOR = {
			has_character_flag = had_event_cfp_artifact_0005
			has_game_rule = never_cfp_artifacts # Disable if rule is OFF
		}
	}
	
	immediate = {
		create_character = {
			gender = male
			age = { 20 60 }
			trait = arrogant
			trait = beauty_bad_2
			location = root.capital_province
			culture = root.culture
			faith = root.faith
			save_scope_as = sculptor
			dynasty = none
		}
		add_character_flag = {
			flag = had_event_cfp_artifact_0005
			days = 7300
		}
	}
	
	## Go away!
	option = {
		name = cfp_artifact.0005.a
		debug_log = "CFP_Debug: a ruler refused to be the benefactor of a local sculptor in cfp_artifact.0005"
		
		stress_impact = {
			zealous = minor_stress_impact_gain
			generous = minor_stress_impact_gain
			trusting = minor_stress_impact_gain
		}
		
		ai_chance = {
			base = 40
			ai_value_modifier = {
				ai_greed = 1
			}
			modifier = {
				gold < 40
				add = -40
			}
			modifier = {
				OR = {
					has_trait = cynical
					has_trait = greedy
				}
				add = 100
			}
			modifier = {
				any_character_artifact = { 
					OR = {
						has_variable = cfp_potato_jesus_var
						has_variable = cfp_saint_statue_var
						has_variable = cfp_mary_statue_var
					}
				}
				add = 20
			}
		}
		
		hidden_effect = {
			scope:sculptor = {
				death = { death_reason = death_vanished	}
			}
		}
	}
	
	## Ok
	option = {
		name = cfp_artifact.0005.b
		
		remove_short_term_gold = { 30 40 }
		
		if = {
			limit = { exists = scope:sculptor }
			add_visiting_courtier = scope:sculptor
		}
		
		stress_impact = {
			greedy = medium_stress_impact_gain
			cynical = medium_stress_impact_gain
		}
		
		ai_chance = {
			base = 10
			ai_value_modifier = {
				ai_zeal = 1
			}
			modifier = {
				has_trait = zealous
				add = 200
			}
		}
		
		hidden_effect = {
		
			random_list = {
				35 = {
					debug_log = "CFP_Debug: sculptor disapeared in cfp_artifact.0006"
					trigger_event = {
						id = cfp_artifact.0006
						days = { 5 10 }
					}
				}
				30 = {
					if = { 
						limit = { NOT = { any_character_artifact = { has_variable = cfp_potato_jesus_var } } } 
						trigger_event = {
							id = cfp_artifact.0007
							days = { 15 30 }
						}
					}
					else = {
						debug_log = "CFP_Debug: sculptor disapeared in cfp_artifact.0006"
						trigger_event = {
							id = cfp_artifact.0006
							days = { 5 10 }
						}
					}
				}
				20 = {
					if = {
						limit = { NOT = { any_character_artifact = { has_variable = cfp_saint_statue_var } } } 
						debug_log = "CFP_Debug: sculptor made a masterwork artifact in cfp_artifact.0008"
						trigger_event = {
							id = cfp_artifact.0008
							days = { 30 60 }
						}
					}
					else = {
						debug_log = "CFP_Debug: sculptor disapeared in cfp_artifact.0006"
						trigger_event = {
							id = cfp_artifact.0006
							days = { 5 10 }
						}
					}
				}
				15 = {
					if = {
						limit = { NOT = { any_character_artifact = { has_variable = cfp_mary_statue_var } } } 
						debug_log = "CFP_Debug: sculptor made a famed artifact in cfp_artifact.0009"
						trigger_event = {
							id = cfp_artifact.0009
							days = { 90 180 }
						}
					}
					else = {
						debug_log = "CFP_Debug: sculptor disapeared in cfp_artifact.0006"
						trigger_event = {
							id = cfp_artifact.0006
							days = { 5 10 }
						}
					}
				}
			}
		}
	}
}

# Your marshal thinks that a chess set would help, and you can prove yourself worthy
cfp_artifact.0011 = {
	type = character_event
	title = cfp_artifact.0011.t
	desc = cfp_artifact.0011.desc
	theme = martial_strategy_focus
	left_portrait = {
		character = scope:marshal
		animation = marshal
	}
	
	trigger = {
		exists = cp:councillor_marshal
		cp:councillor_marshal = { 
			martial >= average_skill_level
			basic_is_valid_for_yearly_events_trigger = yes
			is_performing_council_task = task_train_commanders
		}
		NOR = {
			any_character_artifact = { has_variable = cfp_chess_set_var }
			has_character_flag = had_event_cfp_artifact_0011
			location = { geographical_region = world_africa }
			has_game_rule = never_cfp_artifacts # Disable if rule is OFF
		}
	}
	
	immediate = {
		if = {
			limit = {
				exists = cp:councillor_marshal
			}
			cp:councillor_marshal = { save_scope_as = marshal }
		}
		add_character_flag = {
			flag = had_event_cfp_artifact_0011
			days = 3650
		}
	}
	
	option = {
		name = cfp_artifact.0011.a
		debug_log = "CFP_Debug: a ruler was not interested in a chess set in cfp_artifact.0011"
		
		stress_impact = {
			trusting = minor_stress_gain
			lazy = minor_stress_loss
		}
		
		ai_chance = {
			base = 25
			ai_value_modifier = {
				ai_rationality = -1
			}
		}
		
		scope:marshal = {
			add_opinion = {
				target = root
				modifier = respect_opinion
				opinion = -5
			}
		}
	}
	option = {
		name = cfp_artifact.0011.b
		
		remove_short_term_gold = minor_gold_value
		add_prestige = 15
		
		stress_impact = {
			greedy = minor_stress_impact_gain
		}
		
		add_character_flag = {
			flag = paid_for_cheap_chess_set
			days = 100
		}
		
		ai_chance = {
			base = 50

				gold < minor_gold_value

			}

				OR = {
					has_trait = greedy
					ai_greed >= medium_positive_ai_value
				}
				add = -40
			}
		}
		
		trigger_event = {
			id = cfp_artifact.0012
			days = { 30 90 }
		}
	}
	option = {
		name = cfp_artifact.0011.c
		
		remove_short_term_gold = major_gold_value

			OR = {

			}

			greedy = medium_stress_impact_gain
		}

			base = 100

				gold < major_gold_value
